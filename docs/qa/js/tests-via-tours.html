How to run odoo tours in phantom_js method?

10.0+¶

from odoo.tests.common import HttpCase

class CLASS_NAME(HttpCase):
    def test_NAME(self):

        tour = 'TOUR_NAME'
        self.phantom_js(
            URL_PATH,

            "odoo.__DEBUG__.services['web_tour.tour']"
            ".run('%s')" % tour,

            "odoo.__DEBUG__.services['web_tour.tour']"
            ".tours['%s'].ready" % tour,

            login=LOGIN_OR_NONE
        )
